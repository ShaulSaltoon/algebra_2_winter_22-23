\begin{exercise}
יהי
$V$
מרכב מכפלה פנימית מעל שדה
$\mbb{F} \in \set{\mbb{R}, \mbb{C}}$.
נגיד כי וקטורים
$u,v$
\emph{ניצבים}
אם
$\trs{u,v} = 0$,
ונסמן זאת
$u \perp v$.
עבור תת־קבוצה
$S \subseteq V$
נגדיר את
\emph{המרחב הניצב}
\[\text{.} S^\perp = \set{v \in V}{\forall s \in S : v \perp s}\]

\begin{enumerate}
\item הראו כי
$S^\perp \leq V$
תת־מרחב וקטורי של
$V$.

\item תהי
$S \subseteq V$.
הראו כי
$S \subseteq \prs{S^{\perp}}^{\perp}$
והסיקו כי
$\Span\prs{S} \subseteq \prs{S^{\perp}}^{\perp}$.

\item
נראה בהרצאה כי עבור
$W \leq V$
מתקיים
$V = W \oplus W^\perp$.
היעזרו בכך כדי להראות כי
$\prs{S^\perp}^\perp = \Span\prs{S}$
לכל
$S \subseteq V$.
\end{enumerate}
\end{exercise}

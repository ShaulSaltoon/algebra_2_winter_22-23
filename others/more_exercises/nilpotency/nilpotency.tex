\begin{exercise}
יהי
$V$
מרחב וקטורי ממימד
$n > 0$
ותהי
\[0 < n_1 < n_2 < \ldots < n_{k-1} < n_k = n\]
סדרת מספרים.

\begin{enumerate}
\item נניח כי
$n_i - n_{i-1}$
מונוטונית יורדת (לאו דווקא ממש).
הראו שיש העתקה
$T \in \endo_{\mbb{F}}\prs{V}$
עבורה
\[n_i = \dim\ker\prs{T^i}\]
לכל
$i \in [k]$.

\textbf{רמז:}
חישבו על מטריצות מהצורה
$J_m\prs{0}$.

\item נניח כי יש
$i \in [k]$
עבורו
$n_{i+1} - n_i > n_i - n_{i-1}$.
הראו שאין העתקה
$T \in \endo_{\mbb{F}}\prs{V}$
עבורה
\[n_i = \dim\ker\prs{T^i}\]
לכל
$i \in [k]$.
\end{enumerate}
\end{exercise}
